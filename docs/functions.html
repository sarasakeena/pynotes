# Mini calculator
def calc(a, b, op):
    if op == '+': return a + b
    elif op == '-': return a - b
    elif op == '*': return a * b
    elif op == '/': return a / b
    else: return "Invalid"

calc(10, 5, '*')

50

# Use input + call functions
def main():
    a = int(input("Enter a: "))
    b = int(input("Enter b: "))
    op = input("Enter operator (+ - * /): ")
    print("Result:", calc(a, b, op))

main() 

Enter a:  2
Enter b:  2
Enter operator (+ - * /):  +


Result: 4
