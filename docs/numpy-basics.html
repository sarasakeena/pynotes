Numpy Basics

import numpy as np

a = np.array([1, 2, 3])
b = np.array([[1, 2], [3, 4]])

print("1D array:", a)
print("2D array:\n", b)

1D array: [1 2 3]
2D array:
 [[1 2]
 [3 4]]

print("Sum:", a.sum())
print("Mean:", a.mean())
print("Max:", a.max())
print("Shape of b:", b.shape)

Sum: 6
Mean: 2.0
Max: 3
Shape of b: (2, 2)

print("a[1]:", a[1])
print("b[0, 1]:", b[0, 1])
print("a[1:]:", a[1:])

a[1]: 2
b[0, 1]: 2
a[1:]: [2 3]
